jQuery Favicon Notifier

A super simple plugin that renders a notification number on the browser favicon. Sources on github.

Usage

	// All paramaters after image are optional
	// $.faviconNotify(image[, number, position, outlinecolor, textcolor]);
	// Reset
	$.faviconNotify('/favicon.ico');
	// Number
	$.faviconNotify('/favicon.ico', 12);
	// Custom positions
	$.faviconNotify('/favicon.ico', 12, 'tr');
	$.faviconNotify('/favicon.ico', 12, 'br');
	$.faviconNotify('/favicon.ico', 12, 'tl');
	$.faviconNotify('/favicon.ico', 12, 'bl');
	// Custom icons and colours
	$.faviconNotify('flask.png', 4, 'tr', '#ABCDEF', '#123456');
	$.faviconNotify('pill.png', 9, 'br', '#FFFFFF', '#000000');
